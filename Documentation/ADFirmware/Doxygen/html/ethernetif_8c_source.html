61@Note: This interface is implemented to operate in zero-copy mode only:
62 - Rx buffers will be allocated from LwIP stack memory heap,
63 then passed to ETH HAL driver.
64 - Tx buffers will be allocated from LwIP stack memory heap,
65 then passed to ETH HAL driver.
66
67@Notes:
68 1.a. ETH DMA Rx descriptors must be contiguous, the default count is 4,
69 to customize it please redefine ETH_RX_DESC_CNT in ETH GUI (Rx Descriptor Length)
70 so that updated value will be generated in stm32xxxx_hal_conf.h
71 1.b. ETH DMA Tx descriptors must be contiguous, the default count is 4,
72 to customize it please redefine ETH_TX_DESC_CNT in ETH GUI (Tx Descriptor Length)
73 so that updated value will be generated in stm32xxxx_hal_conf.h
74
75 2.a. Rx Buffers number must be between ETH_RX_DESC_CNT and 2*ETH_RX_DESC_CNT
76 2.b. Rx Buffers must have the same size: ETH_RX_BUFFER_SIZE, this value must
77 passed to ETH DMA in the init field (heth.Init.RxBuffLen)
78 2.c The RX Ruffers addresses and sizes must be properly defined to be aligned
79 to L1-CACHE line size (32 bytes).

786void HAL_ETH_RxLinkCallback(void **pStart, void **pEnd, uint8_t *buff, uint16_t Length)
787{
788/* USER CODE BEGIN HAL ETH RxLinkCallback */
789
790 struct pbuf **ppStart = (struct pbuf **)pStart;
791 struct pbuf **ppEnd = (struct pbuf **)pEnd;
792 struct pbuf *p = NULL;
793
794 /* Get the struct pbuf from the buff address. */
795 p = (struct pbuf *)(buff - offsetof(RxBuff_t, buff));
796 p->next = NULL;
797 p->tot_len = 0;
798 p->len = Length;
799
800 /* Chain the buffer. */
801 if (!*ppStart)
802 {
803 /* The first buffer of the packet. */
804 *ppStart = p;
805 }
806 else
807 {
808 /* Chain the buffer to the end of the packet. */
809 (*ppEnd)->next = p;
810 }
811 *ppEnd = p;
812
813 /* Update the total length of all the buffers of the chain. Each pbuf in the chain should have its tot_len
814 * set to its own length, plus the length of all the following pbufs in the chain. */
815 for (p = *ppStart; p != NULL; p = p->next)
816 {
817 p->tot_len += Length;
818 }
819
820 /* Invalidate data cache because Rx DMA's writing to physical memory makes it stale. */
821 SCB_InvalidateDCache_by_Addr((uint32_t *)buff, Length);
822
823/* USER CODE END HAL ETH RxLinkCallback */
824}
